class Veichel {

private:
    int countOfWheels = 4;
    int countOfDrivers = 1;
    int carSpeed = 0;

public:
    Veichel(string newMark, string newColor, string newDriverName) {
        mark = newMark;
        carColor = newColor;
        driverName = newDriverName;
        // Конструктор с 3мя параметрами
        printCarInformation();
    }
    Veichel(string newMark, string newColor) {
        mark = newMark;
        carColor = newColor;
        driverName = "default";
        // Конструктор с 3мя параметрами
        printCarInformation();
    }
    Veichel(string newMark) {
        mark = newMark;
        carColor = "default";
        driverName = "default";
        // Конструктор с 3мя параметрами
        printCarInformation();
    }
    string mark;
    string carColor;
    string driverName;


    void printCarInformation() {
        cout << "Был создан класс автомобиля:\t" << mark
            << "\nЦвет автомобиля:\t" << carColor
            << "\nИмя водителя:\t" << driverName
            << "\nСкорость автомобиля:\t" << carSpeed << "км/ч\n";
    }

    void printCarInformation(bool status) {
        cout << "Был создан класс автомобиля:\t" << mark
            << "\nКоличество колес:\t" << countOfWheels
            << "\nКоличество водителей:\t" << countOfDrivers
            << "\nСкорость автомобиля:\t" << carSpeed << "км/ч\n";
    }

    void setSpeed(int newSpeedCount) {
        cout << "Изменение скорости с " << carSpeed << " -> " << newSpeedCount << endl;
        carSpeed = newSpeedCount;
    }

    int getSpeed() {
        return carSpeed;
    }

    void setWheelsCount() {
        int count = 0;

        do {
            cout << "Ведите новое количество колес (от 1 до 4): ";
            cin >> count;
        } while (count <= 0 || count >= 5);
        countOfWheels = count;
    }
    int getWheelsCount() {
        return countOfWheels;
    }

    void setCountOfDrivers(int newCount) {
        countOfDrivers = newCount;
    }
    int getCountOfDrivers() {
        return countOfDrivers;
    }
};


// 1а
class Audi : public Veichel {

    Audi(string mark,

    }

        cout << "1 аргумент сработали\n";
        cout << "Количество колес у " << mark << " равняется:\t" << getWheelsCount() << endl;
    }
};
